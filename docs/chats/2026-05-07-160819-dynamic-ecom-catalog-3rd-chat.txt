     v4: Product Variants (Phase 1 вҖ” Size)

     Context

     PR #6 (v3 attribute filters) is merged into main. 171 tests (111 backend,
     60 frontend). 29 seed products.

     Problem: size: 'M' is wrong вҖ” a single shirt listing has sizes S, M, L, XL,
      each with independent stock. The current flat-attribute model cannot
     represent this.

     User decisions confirmed:
     - Variants model (separate stock per size)
     - Phase 1 scope: size only. Color and material planned for Phase 2+.
     - Filter: only show products with in-stock variants matching the selected
     size
     - Product card: show all variant sizes as chips (out-of-stock greyed out)

     ---
     Postgres Target Architecture

     This is the data model the in-memory implementation mirrors 1:1.

     Why not store size: ['S','M','L'] on the product row?

     Storing an array in products.attributes.size (JSONB) means you lose the
     per-size stock. You'd need to add a size_stocks: {"S": 20, "M": 40} map
     alongside it вҖ” ugly and error-prone. Every stock update touches the main
     product row and invalidates its GIN index entries. Concurrent writes
     contend on a single row.

     A product_variants table solves this:
     - Each size is one row вҶ’ stock update is a single-row UPDATE WHERE id = $1
     вҖ” no table-level lock, no JSONB rewrite
     - Filtering "size M, in stock" = JOIN product_variants WHERE options @> 
     '{"size":"M"}' AND stock > 0 вҖ” GIN-indexed
     - Future: adding color = adding rows, not schema migrations

     Tables

     -- Products: identity, base price, non-variant attributes
     CREATE TABLE products (
       id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
       name TEXT NOT NULL,
       description TEXT DEFAULT '',
       price DECIMAL(10,2) NOT NULL,
       currency CHAR(3) NOT NULL,
       category TEXT NOT NULL,
       type TEXT NOT NULL,
       brand_id UUID REFERENCES brands(id),
       tags TEXT[] DEFAULT '{}',
       images TEXT[] DEFAULT '{}',
       attributes JSONB NOT NULL DEFAULT '{}', -- color, material, fit, ram_gb, 
     etc. (non-variant)
       created_at TIMESTAMPTZ DEFAULT NOW()
     );
     CREATE INDEX ON products USING GIN(attributes);   -- scalar attr filtering
     CREATE INDEX ON products (category);
     CREATE INDEX ON products (brand_id);

     -- ProductVariants: one row per available variant
     CREATE TABLE product_variants (
       id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
       product_id UUID NOT NULL REFERENCES products(id) ON DELETE CASCADE,
       options JSONB NOT NULL DEFAULT '{}',  -- {"size": "M"} | Phase 2: 
     {"size": "M", "color": "black"}
       stock INTEGER NOT NULL DEFAULT 0 CHECK (stock >= 0),
       sku TEXT UNIQUE,
       UNIQUE (product_id, options)          -- no duplicate variant per product
     );
     CREATE INDEX ON product_variants USING GIN(options);  -- variant option 
     filtering
     CREATE INDEX ON product_variants (product_id);        -- fetch all variants
      for a product

     -- AttributeRegistry: O(1) schema discovery (no product scan)
     CREATE TABLE attribute_registry (
       category TEXT NOT NULL,
       attribute_key TEXT NOT NULL,
       attribute_type TEXT NOT NULL,              -- 'string' | 'number' | 
     'boolean'
       is_variant_dimension BOOLEAN DEFAULT FALSE, -- true вҶ’ filter via 
     product_variants, not products.attributes
       string_values TEXT[] DEFAULT '{}',
       number_min DECIMAL,
       number_max DECIMAL,
       PRIMARY KEY (category, attribute_key)
     );

     Key read queries

     -- Filter: apparel products with in-stock size M
     SELECT DISTINCT p.*
     FROM products p
     JOIN product_variants pv ON pv.product_id = p.id
       AND pv.options @> '{"size": "M"}'::jsonb
       AND pv.stock > 0
     WHERE p.category = 'apparel';

     -- Filter: size M OR L (bitmap OR of two GIN scans)
     SELECT DISTINCT p.*
     FROM products p
     JOIN product_variants pv ON pv.product_id = p.id
       AND (pv.options @> '{"size": "M"}'::jsonb OR pv.options @> '{"size": 
     "L"}'::jsonb)
       AND pv.stock > 0
     WHERE p.category = 'apparel';

     -- Filter: size M AND color black (Phase 2 вҖ” single @> query)
     SELECT DISTINCT p.*
     FROM products p
     JOIN product_variants pv ON pv.product_id = p.id
       AND pv.options @> '{"size": "M", "color": "black"}'::jsonb
       AND pv.stock > 0;

     -- Scalar attribute filter (electronics, ram_gb >= 16)
     WHERE (p.attributes->>'ram_gb')::int >= 16
     -- Requires expression index (lazy, per category):
     CREATE INDEX ON products (((attributes->>'ram_gb')::int)) WHERE category =
     'electronics';

     -- Fetch product + variants (detail view)
     SELECT p.*, json_agg(json_build_object('id', pv.id, 'options', pv.options,
     'stock', pv.stock)) AS variants
     FROM products p
     LEFT JOIN product_variants pv ON pv.product_id = p.id
     WHERE p.id = $1
     GROUP BY p.id;

     Key write queries

     -- Create product (atomic):
     BEGIN;
       INSERT INTO products (...) VALUES (...) RETURNING id;
       INSERT INTO product_variants (product_id, options, stock) VALUES
         ($id, '{"size": "S"}', 25), ($id, '{"size": "M"}', 40), ...;
       INSERT INTO attribute_registry ... ON CONFLICT (category, attribute_key)
     DO UPDATE ...;
     COMMIT;

     -- Update stock for one variant (hot path вҖ” single row, no JSONB rewrite):
     UPDATE product_variants SET stock = $1 WHERE id = $2;

     -- Add new size option:
     INSERT INTO product_variants (product_id, options, stock) VALUES ($1,
     '{"size": "XL"}', 0);

     -- Remove a variant:
     DELETE FROM product_variants WHERE id = $1 AND product_id = $2;

     Step 3 вҖ” Store: Variant-Aware search()

     In search(), attribute filter block:

     if (attrFilters && typeof attrFilters === 'object') {
       for (const [key, val] of Object.entries(attrFilters)) {
         const catReg = attributeRegistry.get(product.category) || {};
         const isVariantDim = catReg[key]?.isVariantDimension || false;

         if (isVariantDim) {
           // Filter via variants вҖ” must have in-stock variant matching the 
     filter
           const matchingVariants = (product.variants || []).filter(v => {
             const vVal = v.options[key];
             if (vVal == null) return false;
             if (Array.isArray(val)) {
               return val.length === 0 ||
     val.map(String).includes(String(vVal));
             }
             return String(vVal) === String(val);
           }).filter(v => v.stock > 0);  // stock-aware: out-of-stock variants 
     don't match

           if (Array.isArray(val) && val.length === 0) continue; // no filter = 
     pass
           if (matchingVariants.length === 0) return false;

         } else {
           // Scalar attribute filter вҖ” existing logic unchanged
           const productVal = product.attributes[key];
           if (productVal == null) return false;
           if (Array.isArray(val)) {
             if (val.length === 0) continue;
             if (!val.map(String).includes(String(productVal))) return false;
           } else if (val !== null && typeof val === 'object') {
             const numVal = Number(productVal);
             if (isNaN(numVal)) return false;
             if (val.min !== undefined && val.min !== '' && numVal <
     Number(val.min)) return false;
             if (val.max !== undefined && val.max !== '' && numVal >
     Number(val.max)) return false;
           } else {
             if (String(productVal) !== String(val)) return false;
           }
         }
       }
     }

     Step 4 вҖ” Store: Variant CRUD + Stock Recomputation

     New functions in inMemoryStore.js:

     function addVariant(productId, variantData) {
       const product = findById(productId);
       if (!product) return null;
       const variant = { id: nanoid(), options: variantData.options || {},
     stock: variantData.stock || 0 };
       const updated = { ...product, variants: [...(product.variants || []),
     variant] };
       updated.stock = updated.variants.reduce((s, v) => s + v.stock, 0);
       store.set(productId, updated);
       updateRegistry(updated.category, variant.options, true);
       return { ...updated };
     }

     function updateVariant(productId, variantId, patch) {
       const product = findById(productId);
       if (!product) return null;
       const variants = (product.variants || []).map(v =>
         v.id === variantId ? { ...v, ...patch, id: v.id } : v
       );
       const updated = { ...product, variants, stock: variants.reduce((s, v) =>
     s + v.stock, 0) };
       store.set(productId, updated);
       return { ...updated };
     }

     function removeVariant(productId, variantId) {
       const product = findById(productId);
       if (!product) return null;
       const variants = (product.variants || []).filter(v => v.id !==
     variantId);
       const updated = { ...product, variants, stock: variants.reduce((s, v) =>
     s + v.stock, 0) };
       store.set(productId, updated);
       return { ...updated };
     }
